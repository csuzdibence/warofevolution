260  }
261  else
262  {
263  MyProfile newProfile = new MyProfile(this.ProfileName, 0);
264  newProfile.CompletedLevels = 1;
265  newProfile.AddWeapon(new RangedWeapon { Name = "rock", Damage = 10, Price = 20 });
266  newProfile.AddWeapon(new MeleeWeapon { Name = "stick", Damage = 15, Price = 30 });
267  this.Profiles.Add(newProfile);
268  this.IsTextBoxVisible = Visibility.Hidden;
269  this.Message = "Adding OK";
