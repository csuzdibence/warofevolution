120  public bool IsBuyable(MyProfile myProfile, Weapon weapon)
121  {

124  {

153  public bool IsCraftable(MyProfile profile, Weapon weapon)

156  profile.Weapons.Where(x => x.Name == weapon.CraftItemA).Count() > 0

158  {

160  profile.Weapons.Remove(profile.Weapons.Where(x => x.Name == weapon.CraftItemB).FirstOrDefault());
161  return true;
162  }
