73  public void WriteData(ICollection<MyProfile> collection)
74  {

77  {

80  profile.Add(new XElement("gold", item.NumOfGolds));
81  profile.Add(new XElement("profilelevel", item.CompletedLevels));

84  XElement w = new XElement("weapon", string.Empty);
85  string wtype = weapon is MeleeWeapon ? "melee" : "ranged";
86  w.SetAttributeValue("type", wtype);
87  w.Add(new XElement("wname", weapon.Name));
88  w.Add(new XElement("damage", weapon.Damage));
89  w.Add(new XElement("price", weapon.Price));
90  w.Add(new XElement("level", weapon.LevelRequired));
91  profile.Add(w);
