212  private void MoveToLevelCompleted()
213  {
214  this.tickTimer.Stop();
215  this.stw.Stop();
216  double elapsedSeconds = this.stw.ElapsedMilliseconds / 1000;
217  this.currentPage.time.Content = "TIME IN SECONDS: " + elapsedSeconds;
218  double timeRate = (Config.LevelTime - elapsedSeconds) / 100;
219  double bonusGold = this.CalculateBonus(elapsedSeconds, timeRate);
220  int gold = (int)Math.Round(bonusGold, 0);
221  if (gold > 0)
222  {
223  this.logic.AddGold(gold);
224  }
225 
226  this.logic.SaveProfileData(this.levelIndex);
227  this.currentPage.bonusGold.Content = "BONUS GOLDS: " + gold;
228  this.currentPage.gameCompletedMenu.Visibility = Visibility.Visible;
229  }
