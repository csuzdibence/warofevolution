52  new XElement("profilelevel", profileLevel));

55  XElement weapon = new XElement(
56  "weapon",
57  new XElement("wname", item.Name),
58  new XElement("damage", item.Damage),
59  new XElement("price", item.Price),

63  weapon.SetAttributeValue("type", "ranged");

70  profileData.Add(weapon);

73  xdoc.Descendants("profile")?.Where(x => x.Element("name")?.Value == profile.Name).Remove();

132  int weaponOffsetY = int.Parse(xweapon.Attribute("offsetY")?.Value);
133  int weaponWidth = int.Parse(xweapon.Attribute("width")?.Value);

135  string image = xweapon.Element("image")?.Value;
136  if (xweapon.Attribute("type")?.Value == "melee")
137  {
138  gameModel.Player.Weapons.Add(new MeleeWeapon(
139  Config.PlayerStartingX + weaponOffsetX,
140  (int)gameModel.GameHeight + weaponOffsetY - Config.GroundY - (Config.PlayerHeight / 2),
141  weaponWidth,
142  weaponHeight)
143  { Damage = w.Damage, Name = w.Name, Price = w.Price, Image = image, OffsetX = weaponOffsetX, OffsetY = weaponOffsetY });
144  }
145  else
146  {
147  gameModel.Player.Weapons.Add(new RangedWeapon(
148  Config.PlayerStartingX + weaponOffsetX,
149  (int)gameModel.GameHeight + weaponOffsetY - Config.GroundY - (Config.PlayerHeight / 2),
150  weaponWidth,
151  weaponHeight)
152  { Damage = w.Damage, Name = w.Name, Price = w.Price, Image = image, OffsetX = weaponOffsetX, OffsetY = weaponOffsetY });
153  }

161  private void BuildLevel(GameModel gameModel, int levelNumber)
162  {
163  XDocument xdoc = XDocument.Load(this.levelXml);
164  XElement selectedLevel = xdoc.Descendants("level")?.Where(x => int.Parse(x.Element("number")?.Value) == levelNumber).FirstOrDefault();
165  Level level = new Level();
166  level.BackgroundImage = selectedLevel.Element("backgroundimage")?.Value;
167  level.ObstacleImage = selectedLevel.Element("obstacle")?.Value;
168  level.LevelTitle = selectedLevel.Element("title")?.Value;
169  int randScenes = rnd.Next(Config.MinNumOfScenes, Config.MaxNumOfScenes + 1);
170  for (int i = 0; i < randScenes; i++)
171  {
172  Scene scene = new Scene();
173  int randObstacles = rnd.Next(Config.MinNumOfObstacles, Config.MaxNumOfObstacles + 1);
174 
175  for (int j = 0; j < randObstacles; j++)
176  {
177  int randX = rnd.Next(((int)gameModel.GameWidth / 5) + (j * 50), (int)gameModel.GameWidth - (5 * Config.ObstacleWidth));
178  int randY = rnd.Next(((int)gameModel.GameHeight / 2) + (j * 50), (int)gameModel.GameHeight - (2 * Config.ObstacleHeight));
179  scene.Obstacles.Add(new Obstacle(randX, randY, Config.ObstacleWidth, Config.ObstacleHeight));
180  }
181 
182  level.Scenes.Add(scene);
183  }
184 
185  level.Portal = new Portal(
186  (int)gameModel.GameWidth - Config.PortalSize - 50,
187  (int)gameModel.GameHeight - Config.GroundY - (Config.PortalSize - Config.PlayerHeight),
188  Config.PortalSize,
189  Config.PortalSize);
190 
191  gameModel.Enemies = new List<Enemy>();
192  level.AvailableEnemyWeapons = new List<Weapon>();
193  foreach (var w in selectedLevel.Descendants("weapon"))
194  {
195  int weaponOffsetX = int.Parse(w.Attribute("offsetX")?.Value);
196  int weaponOffsetY = int.Parse(w.Attribute("offsetY")?.Value);
197  int weaponWidth = int.Parse(w.Attribute("width")?.Value);

199  string image = w.Element("image")?.Value;
200  if (w.Attribute("type")?.Value == "melee")
201  {
202  level.AvailableEnemyWeapons.Add(new MeleeWeapon(0, 0, weaponWidth, weaponHeight)
203  {
204  Damage = int.Parse(w.Element("damage")?.Value),
205  Name = w.Element("wname")?.Value,
206  Image = image,
207  OffsetX = weaponOffsetX,
208  OffsetY = weaponOffsetY,
209  });
210  }
211  else
212  {
213  level.AvailableEnemyWeapons.Add(new RangedWeapon(0, 0, weaponWidth, weaponHeight)
214  {
215  Damage = int.Parse(w.Element("damage")?.Value),
216  Name = w.Element("wname")?.Value,
217  Image = image,
218  OffsetX = weaponOffsetX,
219  OffsetY = weaponOffsetY,
220  });
221  }
222  }
223 
224  // Build Enemies and levelInfo

226  int enemyHeight = int.Parse(selectedLevel.Element("enemyimage")?.Attribute("height")?.Value);
227  int enemyWidth = int.Parse(selectedLevel.Element("enemyimage")?.Attribute("width")?.Value);
228  int enemyHealth = int.Parse(selectedLevel.Element("enemyhealth")?.Value);
